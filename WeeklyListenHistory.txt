/* Historical record of WeeklyListenDB
Create Database "WeeklyListenDB"*/
CREATE DATABASE WeeklyListenDB;
USE WeeklyListenDB;
/* Create User Table */
CREATE TABLE user (
user_id SMALLINT NOT NULL auto_increment,
firstname VARCHAR(100) NOT NULL,
lastname VARCHAR(100) NOT NULL,
/* Let's make fach a separate table
fach ENUM('Coloratura Soprano', 'Soubrette', 'Lyric Soprano', 'Spinto Soprano', 'Dramatic Soprano', 
'Coloratura Mezzo-Soprano', 'Lyric Mezzo-Soprano', 'Dramatic Mezzo-Soprano', 'Contralto',
'Countertenor', 'Leggero Tenor', 'Lyric Tenor', 'Spinto Tenor', 'Character Tenor', 'Heldentenor',
'Lyric Baritone', 'Kavalierbariton', 'Dramatic Baritone', 'Bass-Baritone',
'Bass Buffo', 'Basso profondo') DEFAULT NULL,
*/
PRIMARY KEY (user_id)
) ENGINE = InnoDB DEFAULT CHARSET=latin1;

/*Create "Posts" Table*/
CREATE TABLE post (
post_id SMALLINT NOT NULL auto_increment,
user_id SMALLINT NOT NULL,
link VARCHAR(2083) NOT NULL,
firstname VARCHAR(100) NOT NULL,
lastname VARCHAR(100) NOT NULL,
primary_fach ENUM('Soprano', 'Mezzo-Soprano','Contralto','Countertenor','Tenor','Baritone','Bass') NOT NULL,
secondary_fach VARCHAR(100) DEFAULT NULL,
/*singer_id SMALLINT NOT NULL,
aria_id SMALLINT NOT NULL,
opera_id SMALLINT NOT NULL,
composer_id SMALLINT NOT NULL,*/
PRIMARY KEY (post_id)
) ENGINE = InnoDB DEFAULT CHARSET=latin1;
